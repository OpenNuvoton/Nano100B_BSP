173 void SendNEC(uint8_t* data)
174 {
175  uint8_t nbyte;
176  uint8_t nbit;
177 
178  /* Send out Leader code */
179  Mark(NEC_LDC_MARK);
180  SPACE(NEC_LDC_SPACE);
181 
182  /* Send out Customer code and Data code */
183  for (nbyte=0; nbyte < NEC_BYTES; nbyte++)
184  {
185  for (nbit=0; nbit < 8; nbit++)
186  {
187  Mark(NEC_BIT_MARK);
188  if (data[nbyte] & (1 << nbit)) // LSB first
189  SPACE(NEC_ONE_SPACE);
190  else
191  SPACE(NEC_ZERO_SPACE);
192  }
193  }
194 
195  /* Send out Stop bit */
