92 void TIMER_Delay(TIMER_T *timer, uint32_t u32Usec)
93 {

95  uint32_t u32Prescale = 0, delay = SystemCoreClock / u32Clk;
96  long long u64Cmpr;

99  timer->CTL = 0;
100

102  {
103  u32Usec = ((u32Usec + 99) / 100) * 100;
104  }
105  else // 10 usec every step
106  {
107  u32Usec = ((u32Usec + 9) / 10) * 10;
108  }
109 
110  if(u32Clk >= 0x2000000)
111  {
112  u32Prescale = 3; // real prescaler value is 4
113  u32Clk >>= 2;
114  }
115  else if(u32Clk >= 0x1000000)
116  {
117  u32Prescale = 1; // real prescaler value is 2
118  u32Clk >>= 1;
119  }
120 
121  // u32Usec * u32Clk might overflow if using uint32_t
122  u64Cmpr = ((long long)u32Usec * (long long)u32Clk) / (long long)1000000;
123 
124  timer->CMPR = (uint32_t)u64Cmpr;
125  timer->PRECNT = u32Prescale;
126  timer->CTL = TIMER_CTL_TMR_EN_Msk; // one shot mode
127

129  // And the while loop below return immediately, so put a tiny delay here allowing timer start counting and raise active flag.
130  for(; delay > 0; delay--)
