266 void USBD_StandardRequest(void)
267 {
268  /* clear global variables for new request */
269  g_usbd_CtrlInPointer = 0;
270  g_usbd_CtrlInSize = 0;
271 
272  if (g_usbd_SetupPacket[0] & 0x80) /* request data transfer direction */
273  {
274  // Device to host
275  switch (g_usbd_SetupPacket[1])
276  {
277  case GET_CONFIGURATION:
278  {
279  // Return current configuration setting
280  /* Data stage */
281  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = g_usbd_UsbConfig;
282  USBD_SET_DATA1(EP0);
283  USBD_SET_PAYLOAD_LEN(EP0, 1);
284  /* Status stage */
285  USBD_PrepareCtrlOut(0,0);
286  break;
287  }
288  case GET_DESCRIPTOR:
289  {
290  USBD_GetDescriptor();
291  break;
292  }
293  case GET_INTERFACE:
294  {
295  // Return current interface setting
296  /* Data stage */
297  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = g_usbd_UsbAltInterface;
298  USBD_SET_DATA1(EP0);
299  USBD_SET_PAYLOAD_LEN(EP0, 1);
300  /* Status stage */
301  USBD_PrepareCtrlOut(0,0);
302  break;
303  }
304  case GET_STATUS:
305  {
306  // Device
307  if(g_usbd_SetupPacket[0] == 0x80)
308  {
309  uint8_t u8Tmp;
310 
311  u8Tmp = 0;
312  if(g_usbd_sInfo->gu8ConfigDesc[7] & 0x40) u8Tmp |= 1; // Self-Powered/Bus-Powered.
313  if(g_usbd_sInfo->gu8ConfigDesc[7] & 0x20) u8Tmp |= (g_usbd_RemoteWakeupEn << 1); // Remote wake up
314 
315  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = u8Tmp;
316  }
317  // Interface
318  else if (g_usbd_SetupPacket[0] == 0x81)
319  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = 0;
320  // Endpoint
321  else if (g_usbd_SetupPacket[0] == 0x82)
322  {
323  uint8_t ep = g_usbd_SetupPacket[4] & 0xF;
324  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = USBD_GetStall(ep)? 1 : 0;
325  }
326 
327  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0) + 1) = 0;
328  /* Data stage */
329  USBD_SET_DATA1(EP0);
330  USBD_SET_PAYLOAD_LEN(EP0, 2);
331  /* Status stage */
332  USBD_PrepareCtrlOut(0,0);
333  break;
334  }
335  default:
336  {
337  /* Setup error, stall the device */
338  USBD_SET_EP_STALL(EP0);
339  USBD_SET_EP_STALL(EP1);
340  break;
341  }
342  }
343  }
344  else
345  {
346  // Host to device
347  switch (g_usbd_SetupPacket[1])
348  {
349  case CLEAR_FEATURE:
350  {
351  if(g_usbd_SetupPacket[2] == FEATURE_ENDPOINT_HALT)
352  {
353 
354  int32_t epNum, i;
355 
356  /* EP number stall is not allow to be clear in MSC class "Error Recovery Test".
357  a flag: g_u32EpStallLock is added to support it */
358  epNum = g_usbd_SetupPacket[4] & 0xF;
359  for(i = 0; i < USBD_MAX_EP; i++)
360  {
361  if(((USBD->EP[i].CFG & 0xF) == epNum) && ((g_u32EpStallLock & (1 << i)) == 0))
362  USBD->EP[i].CFG &= ~(USBD_CFG_SSTALL_Msk | USBD_CFG_DSQ_SYNC_Msk);
363  }
364  }
365  else if(g_usbd_SetupPacket[2] == FEATURE_DEVICE_REMOTE_WAKEUP)
366  g_usbd_RemoteWakeupEn = 0;
367  /* Status stage */
368  USBD_SET_DATA1(EP0);
369  USBD_SET_PAYLOAD_LEN(EP0, 0);
370  break;
371  }
372  case SET_ADDRESS:
373  {
374  g_usbd_UsbAddr = g_usbd_SetupPacket[2];
375 
376  // DATA IN for end of setup
377  /* Status Stage */
378  USBD_SET_DATA1(EP0);
379  USBD_SET_PAYLOAD_LEN(EP0, 0);
380  break;
381  }
382  case SET_CONFIGURATION:
383  {
384  g_usbd_UsbConfig = g_usbd_SetupPacket[2];
385 
386  if (g_usbd_pfnSetConfigCallback)
387  g_usbd_pfnSetConfigCallback();
388 
389  if (g_usbd_UsbConfig == 0)
390  {
391  int volatile i;
392 
393  /* Reset PID DATA0 */
394  for (i = 2; i < USBD_MAX_EP; i++)
395  USBD->EP[i].CFG &= ~USBD_CFG_DSQ_SYNC_Msk;
396  }
397 
398  // DATA IN for end of setup
399  /* Status stage */
400  USBD_SET_DATA1(EP0);
401  USBD_SET_PAYLOAD_LEN(EP0, 0);
402  break;
403  }
404 
405  case SET_FEATURE:
406  {
407  if (g_usbd_SetupPacket[2] == FEATURE_ENDPOINT_HALT)
408  USBD_SetStall(g_usbd_SetupPacket[4] & 0xF);
409  else if (g_usbd_SetupPacket[2] == FEATURE_DEVICE_REMOTE_WAKEUP)
410  g_usbd_RemoteWakeupEn = 1;
411 
412  /* Status stage */
413  USBD_SET_DATA1(EP0);
414  USBD_SET_PAYLOAD_LEN(EP0, 0);
415  break;
416  }
417 
418  case SET_INTERFACE:
419  {
420  g_usbd_UsbAltInterface = g_usbd_SetupPacket[2];
421 
422  if (g_usbd_pfnSetInterface != NULL)
423  g_usbd_pfnSetInterface(g_usbd_UsbAltInterface);
424 
425  /* Status stage */
426  USBD_SET_DATA1(EP0);
427  USBD_SET_PAYLOAD_LEN(EP0, 0);
428  break;
429  }
430 
431  default:
432  {
433  /* Setup error, stall the device */
434  USBD_SET_EP_STALL(EP0);
435  USBD_SET_EP_STALL(EP1);
436  break;
437  }
438  }
439  }

494  g_usbd_CtrlInSize -= g_usbd_CtrlMaxPktSize;

496  else

503  }
504  }
