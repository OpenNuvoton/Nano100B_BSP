34uint32_t I2C_Open(I2C_T *i2c, uint32_t u32BusClock)

90{
91 uint32_t u32Reg = 0;
92 uint32_t u32Val = i2c->CON & ~(I2C_STA | I2C_STO | I2C_AA);
93
94 if (u8Start)
95 u32Reg |= I2C_STA;
96 if (u8Stop)
97 u32Reg |= I2C_STO;
98 if (u8Si)
99 u32Reg |= I2C_SI;
100 if (u8Ack)
101 u32Reg |= I2C_AA;
102
103 i2c->CON = u32Val | u32Reg;
